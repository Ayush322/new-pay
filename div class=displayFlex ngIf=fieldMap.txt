 // getting data by using category codes
  bindMiscData(category, data) {
    switch (category) {
        case CodeCategoryConstants.TIN_TYPE:
this.taxtypes = data.filter(data => data.name === MB.CONST.CREDENTIALS.TAXID);
            break;

        case CodeCategoryConstants.STATE_CODE:
            this.statecodes = data;
            break;

        case CodeCategoryConstants.ACCOUNT_WITHHOLD:
            this.withHold = data;
            break;

        case CodeCategoryConstants.IRA_BENEFICIARY_TYPE:
            this.relationships = data;
            break;
   
        // Combine IRA plan types into a single case
        case CodeCategoryConstants.TRADITIONAL_IRA_TYPE:
        case CodeCategoryConstants.EDUCATIONAL_IRA_TYPE:
        case CodeCategoryConstants.ROTH_IRA_TYPE:
            // Assign the planType based on the current product category
            if (this.currentProduct?.category === "Traditional IRA's") {
                this.planType = data.filter(plan => plan.category === CodeCategoryConstants.TRADITIONAL_IRA_TYPE);
            } else if (this.currentProduct?.category === "Educational IRA's") {
                this.planType = data.filter(plan => plan.category === CodeCategoryConstants.EDUCATIONAL_IRA_TYPE);
            } else if (this.currentProduct?.category === "Roth IRA's") {
                this.planType = data.filter(plan => plan.category === CodeCategoryConstants.ROTH_IRA_TYPE);
            }
            break;
    }
