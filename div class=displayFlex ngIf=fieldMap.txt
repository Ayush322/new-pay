  /* change Titling Opt */
  changeTitlingOpt(event) {
    let titlingID = event.value;
    if (titlingID) {
      this.busyIndicator.show();
      if (this.acctRelationship?.length) {
        this.acctRelationship.forEach((acct) => {
          if (acct.customers) {
            Array.prototype.push.apply(this.unassignedArr, acct.customers);
            acct.customers = [];
          }
        });
      }
      this.applicationService.getAcctRelationship(titlingID).subscribe((data) => {
        this.busyIndicator.hide();
        if (data?.result?.length) {
          this.acctRelationship = this.getBusinessFilteredAcctRelationship(data.result);
          this.acctRelationship.forEach((acct) => {
            acct.customers = [];
          });
        }
      });
    }
  }
