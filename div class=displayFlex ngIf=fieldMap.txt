 taxIdChange(event: MatSelectChange, index: number) {
    const ben = this.beneficiaryArray;
    ben.controls[index].get('beneficiaryTaxId');
    if (event.value === BankerOpenAccountConstants.NOTAXID) {
      this.checkTaxType(index);
    }
  }
  checkTaxType(index: number) {
    if (this.beneficiaryArray.at(index).get('taxIdType').value !== BankerOpenAccountConstants.NOTAXID) {
      this.beneficiaryArray.at(index).get('beneficiaryTaxId').enable();
      return true;
    } else {
      this.beneficiaryArray.at(index).get('beneficiaryTaxId').disable();
      return false;
    }
  }



 bindMiscData(category, data) {
    switch (category) {
       
        case CodeCategoryConstants.IRA_BENEFICIARY_TYPE:
            this.relationships = data;
            break;

            }
}
